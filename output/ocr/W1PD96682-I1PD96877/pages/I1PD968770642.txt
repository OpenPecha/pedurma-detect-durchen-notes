བཀའ་འགྱུར། རྒྱུད་འབུམ། ༼༽ ༩༤
རྣམས་ནི་རྟག་ཏུ་འཕེལ། །ནད་མེད་པ་ཡང་ཐོབ་∶པར་འགྱུར། །●ཕྱིར་ཟློག་པ་
【འཕགས་པ་∶རྣམ་པར་རྒྱལ་བ་•ཅན་ཞེས་བྱ་བ་རྫོགས་སོ∶།།
བསྡུར་མཆན།
༦༡༤ ●མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ●«གཡུང་»ཡ་བ་ནཱིན། 《ལི་»ཡ་
བ་ཏེ་ན། 《སྣར་»ཨ་བ་ཏི་ན། ●«གཡུང་»«ལི་》《པེ་》《ཅོ་»ར། ●«སྣར་»«ཞོལ་»+འཕགས་པ།
●«སྣར་»བཟློག ●«སྣར་»«ཞོལ་»-འཕགས་པ། ●«གཡུང་»བྱ་བའོ། «ཅོ་»བྱའོ། _ ●«ལི་»
《སྣར་»བསྲུང་། ●《སྣར་》«ཞོལ་»སྒྲུབ། ●«ལི་>«སྣར་»བཟློག ●«ལི་》རྣཾས། ●«སྣར་»«ཞོལ་»
+ཐམས་ཅད།
༦༡༥ ●«ཞོལ་»དྭགས། ●«ལི་》《སྣར་»རྗེད། ●«གཡུང་»+དང་། ●«ཞོལ་»+པར་བྱེད། 《སྣར་»
བཟློག་པར་བྱེད། ●«སྣར་»«ཞོལ་»_རབ་ཏུ། ●«གཡུང་»པས། ●«གཡུང་»«པེ་»«ཅོ་»-ཀྱི།
●«ཅོ་»-འདི། ●«སྣར་»«ཞོལ་»+སྟེམས། ●●«ཞོལ་》སྲུང་། ●«ལི་»«ཞོལ་»སྲུང་། ●«སྣར་»
«ཞོལ་»སྲུང་།
༦༡༦ ●«ཞོལ་»-ཧི་ལི། ●«སྣར་»«ཞོལ་»–ཧུ་ལུ། ●«ལི་»བཟློགས། «ཞོལ་»ཟློག ●«སྣར་>«ཞོལ་»
+ན་ཤ་ན་ཤ «ཅོ་»+ནཱ་ཤ་ནཱ་ཡ། ●●«སྣར་》《ཞོལ་»ན། ●«གཡུང་»«པེ་»བསྲེག ●«ཅོ་»རངས།
●«སྣར་»«ཞོལ་»ཞིག ●«ཁུ་»-ཀྱི། ●«གཡུང་》«པེ་»ཏ། ●«གཡུང་»«ལི་»«པེ་>«ཁུ་»གསལ།
●«པེ་»ནམ། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»-རྣམ་པར་གཞིག་ཏུ་གསོལ། ●«སྣར་»ཏྲུ་ཥུས། 《ཅོ་»
ཧཱུྃ༑ ●«གཡུང་>«ལི་>«པེ་》《སྣར་》«ཅོ་»«ཞོལ་»–ཏུ་རུ་ཏུ་རུ། ཏུ་རུ། ●«གཡུང་»«ལི་»«པེ་»
«ཅོ་»–བྱད་དང་། ●«ལི་》《སྣར་》《ཞོལ་»རྗེད། ●«ཞོལ་»སྲུང་། ●«ལི་»«སྣར་»«ཞོལ་»སྲུང་།
●«གཡུང་»«ལི་》«པེ་》«ཅོ་»–དྷུ་རུ་དྷུ་རུ་སྭཱ་ཧཱ། 《སྣར་»«ཞོལ་»-དྷུ་རུ་དྷུ་རུ་སྭཱ་ཧཱ། དྷུ་རུ་དྷུ་རུ། དྷུ་
རུ༑ ●«སྣར་»«ཞོལ་»པ།
༦༡༧ ●«ལི་»བལྟ། ●«གཡུང་»«ཅོ་»འཁྲུལ་བར། «པེ་»འཁྲུག་པར། ●«གཡུང་»«ལི་»«པེ་》《ཅོ་》
-རྨོངས་པར་བགྱིའོ། ●●«གཡུང་»«ཅོ་»འཇིག ●«གཡུང་»«ལི་»«པེ་»མཐོར། ●«ལི་»འཇོ།
●«ཅོ་»ལི། ●«གཡུང་»«པེ་»–ལེ། «ཅོ་»ལི། ●«ཞོལ་»ལེ། ●«སྣར་»«ཞོལ་»ཏི། ●«སྣར་»
«ཅོ་»«ཞོལ་》ཏི། ●«གཡུང་》《པེ་》«ཅོ་》བིཏ་ཀེ། 《སྣར་»«ཞོལ་»པཱི་ཏི་ཀི། ●«སྣར་»ཏྲེ།
●«ལི་»«པེ་»ལོ་ཧི་ཏེ། 《སྣར་»«ཞོལ་»–ལོ་ཧི་ཏ། ●«ལི་»«ཅོ་»ཨྰ། ●«སྣར་»«ཞོལ་»དཱ།
●«ལི་»ཨཱ། ●«སྣར་»«ཞོལ་»དཱ། ●«གཡུང་》«ལི་»«པེ་》ཨཱ། ●《སྣར་》དྷ་ཏ། 《ཅོ་»ད་ཏ།
