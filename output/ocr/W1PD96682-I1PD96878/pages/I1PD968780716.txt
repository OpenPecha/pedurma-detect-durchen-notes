བཀའ་འགྱུར། རྒྱུད་འབུམ། ༼༽ ༩༥
《སྣར་》རྩེ། ●《སྣར་》«ཞོལ་»-རིག་སྔགས།
༥༥༧ ●«གཡུང་»«པེ་》རྗོད། ●«སྣར་»བསྟན། ●«གཡུང་»«ལི་»«པེ་》«ཅོ་»-རྡོ་རྗེ། ●«སྣར་»
«ཞོལ་»སྡེའི་བདག་པོ། ●«གཡུང་»«ལི་»«པེ་》《སྣར་》«ཅོ་»བྱུ། ●«སྣར་»ཀྱི། ●«སྣར་»+བརྒྱ།
●«གཡུང་»«པེ་»གཏང་། «ལི་»བཏང་། 《ཅོ་»མ་ཏང་། ●«སྣར་»-ལ། ●«གཡུང་»«པེ་»
མངོན་གྱིས། ●«པེ་»«ཅོ་»མོའི། ●«གཡུང་»ཡིད་རངས་ཏེ། 《ཞོལ་»ཡི་རང་སྟེ། ●«སྣར་»«ཞོལ་»
-རྒྱ་གར་གྱི། ●«གཡུང་»མཁོན། ●«གཡུང་»«ལི་》«པེ་》《སྣར་》《ཅོ་»«ཞོལ་»ལོཙྪ། ●«སྣར་»
«ཞོལ་»བནྟེ། ●«གཡུང་》སྒྱུར། ●«གཡུང་»+ཞུས་གསུམ་ཞུས་དག། །།ཡེ་དྷརྨཱ་ཧེ་དུ་བྲ་བྷ་བཱ་ཧེ་དུན་ཏེ་
ཥཱན་ཏ་ཐཱ་ག་ཏོ་ཧྱ་བཱ་དད་ཏེ། །ཥཱཉྩ་ཡོ་ན་རི་དྷོ་ཨེ་བཾ་བཱ་དཱི་མ་ཧཱ་ཤྲ་མ་ཎཿ།། «ལི་>«ཅོ་»+ཅིག་ཞུས།
གཉིས་ཞུས། ལན་གསུམ་ཞུས་ཆེན་ཐང་རྒྱལ་གཉིས་ཀྱིས་བགྱིས། །སྟག་ལུང་རྒྱུད་འབུམ་ལ་བསྟུན་ཞུས་དག། །།
«པེ་>+ཞུས་གསུམ་ཞུས་དག། །།ཡེ་དྷརྨཱ་ཧོ་ཏུ་པྲ་བྷ་བཱ་ཧེ་ཏུན་ཏེ་ཥཱན་ཏ་ཐཱ་ག་ཏོ་ཧྱ་བ་དཏ།
དྷ་ཨེ་ཝཾ་བཱ་དཱི་མ་ཧཱ་ཤྲ་མ་ཎཿ།། «ཅོ་»+ཅིག་ཞུས། ཡང་གཉིས་ཞུས། ལན་གསུམ་ཞུས་ཆེན་ཐང་རྒྱལ་གཉིས་
