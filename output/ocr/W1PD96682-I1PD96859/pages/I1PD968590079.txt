བསོད་ནམས་ཀྱི་སྟོབས་ཀྱི་རྟོགས་པ་བརྗོད་པ།
७१
«པེ་»བཅུ། ●«སྣར་»གྱི་ལྷ། «ཁུ་»–པ། ●®●«གཡུང་»«ལི་»«པེ་»བཅུ། ●«ལི་»«ཅོ་»-ཀྱི་
དབང་ཕྱུག ●«གཡུང་»«ལི་»«པེ་»བཅུ། ●«སྣར་》སྟོང་། ●«གཡུང་»«ལི་»«པེ་》«ཅོ་»«ཞོལ་»
བསྒྱུར།༑ ●«ཅོ་»པའི། ●«སྣར་»-རྣམ་པ། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»+འདི། ●«གཡུང་»
«པེ་»+མཆོག ●«སྣར་»+པོ།
༥༡ ●«སྣར་»འཇོམ། ●«སྣར་»«ཞོལ་»དེ་ནས། ●«སྣར་»གནོང་བར། ●«ཞོལ་»མཐུན། ●«གཡུང་»
«པེ་»«སྣར་»《ཞོལ་»ཆེན། ●«གཡུང་»«ལི་»«པེ་»«སྣར་»དང་བར། ●«སྣར་»བསྟི། «ཞོལ་»སྟིར།
●«སྣར་»«ཞོལ་»རི་མོར།། ●«ཞོལ་»པ། ●«སྣར་»བསྟི། «ཞོལ་»སྟིར། ●«སྣར་»-རིམ་གྲོ་བྱེད།
●«གཡུང་»«ལི་«པེ་》«ཅོ་»པར། ●«སྣར་»«ཞོལ་»རྟེན། ●«ལི་»«ཅོ་»-དེ། ●«གཡུང་»«པེ་»
སྟོ།༑ ●«གཡུང་»-རྒྱ་གར་གྱི་མཁན་པོ་......བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པ། «ལི་»རྒྱ་གར་གྱི་མཁན་པོ་འཛི་ན་མི་དྲ་
དང་ཞུ་ཆེན་གྱི་ལོཙྪ་བ་བན་དེ་དེ་བ་ཙན་དྲས་བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པ།
དང་ཞུ་ཆེན་གྱི་ལོཙྪ་བ་བན་དྷེ་དེ་བ་ཙན་དྲས་བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པ།
«པེ་»རྒྱ་གར་གྱི་མཁན་པོ་ཛི་ན་མི་དྲ་
《སྣར་»རྒྱ་གར་གྱི་མཁན་པོ་འཛིན་ན་
མི་ཏྲ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙ་བ་བན་དྷེ་དེ་བ་ཙན་དྲས་བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པའོ།། 《ཅོ་»རྒྱ་གར་གྱི་མཁན་པོ་
ཛི་ན་མི་དྲ་དང་ཞུ་ཆེན་གྱི་ལོཙྪ་བ་བནྡེ་དེ་བ་ཙན་དྲས་བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པ།
ན་མི་ཏྲ་དང་ཞུས་ཆེན་གྱི་ལོ་ཙཱ་བ་བནྡེ་དེ་བ་ཙནྡྲས་བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པའོ།།
