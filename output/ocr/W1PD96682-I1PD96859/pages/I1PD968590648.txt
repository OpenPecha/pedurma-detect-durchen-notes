བ་བཅའ། རིལ་གྱིས།
《ཞོལ་»ལའང་ཁྱེའུ་ལྔ་བརྒྱ་བཙའ་སྟེ། རིལ་གྱིས། ●«གཡུང་»«སྣར་»ཁྲིག་མ།
«ཞོལ་»མཁྲིག་མ། ●«གཡུང་>«པེ་»གྱུར་སྟེ་.... ཅིག་སྐྱེས་ཏེ། «ལི་»བྱུང་སྟེ། དེའི་དུས་ན་རྒྱལ་པོའི་རྟ་
ལ་རྟེའུ་སྨྲ་ཤེས་པ་ཞིག་སྐྱེས་ཏེ། «སྣར་»འབྱུང་དེ་དེང་དུས་གཅིག་ཏུ། «ཅོ་»བྱུང་སྟེ། དེའི་དུས་ན་རྒྱལ་
པོའི་རྟ་ལ་རྟེའུ་སྨྲ་ཤེས་བཞིག་སྐྱེས་ཏེ། ●«སྣར་»-བློན་པོའི་བུ་དང་། «ཞོལ་»བློན་པོ་དང་། ●«སྣར་»
ཅིག་ཀྱང་འབྱུང་སྟེ། དེའི། ●«སྣར་»གནམ། ●«སྣར་»-པ། ●«གཡུང་»«པེ་»གྲ་སེ། 《སྣར་》
བར་སེ། «ཅོ་»དྲསི། ●«སྣར་»གི་འཁོར་དུ། ●«སྣར་»-ལ། ●«སྣར་»-བ་དང་། ●《སྣར་》
-པ༑ ●«གཡུང་»«པེ་»འདི་ཅིའི་བལྟས་ཞེས་སྨྲས་པ་དང་། 《སྣར་»ཅིའི་ལྟས་ཤེས་བརྟགས་པ་ལྷས།
༦༢༢ ●«གཡུང་»«པེ་》བཞི། ●«སྣར་»བྱས་ནས། ●«སྣར་»-ཁྱོད་ཀྱི་བུ། «ཅོ་»–བུ། ●«སྣར་»-རོ།
●«སྣར་»–དེ་ཐོག་མ། ●«སྣར་»དང་། བ་ལཱ་བ་དང་། ●«སྣར་»འདྲ། ●«སྣར་»པོ། ●«ཅོ་»
མིས༑ ●«སྣར་»ལས། ●«སྣར་»པྲ། ●«སྣར་»-ལ། ●«པེ་»ཡག་འདི་ལྟར། «སྣར་»ཡབ་འདི་ལྟ་
●«གཡུང་»«པེ་»བགྱེས། ●«སྣར་»ལས། ●«སྣར་»+བྱིན། ●«སྣར་»-བྱིན་ཆེན་པོ།
●«སྣར་»སོ། ●«གཡུང་»«པེ་»བགྱེས། ●«གཡུང་»«པེ་»བགྱི། «སྣར་»-བགྱིས། ●«སྣར་»རྒྱལ་
པོ༑ ●«གཡུང་»«པེ་»བཅས་ཏེ་བཏང་བ་ཐབ། 《སྣར་»དང་ཆས་ནས་ཐབ། «ཞོལ་»བཅས་ཏེ་བཏང་
ནས་འཐབ། ●«སྣར་»ནས། ●«གཡུང་»«པེ་»རྨེ། ●«སྣར་»-ཡོད་པ། ●«སྣར་»+གྱིས།
●«པེ་»ནས་བརྒོལ། 《སྣར་》ཏེ་བརྒལ། ●«སྣར་»ཧའི། ●«ཞོལ་»འཐབ། ●«ལི་»«པེ་»«ཅོ་》
ཅད༑ ●«ཞོལ་»ཟུངས། ●«གཡུང་»-ཀྱང་། 《སྣར་»བུས་བཟུང་། ●«གཡུང་»«པེ་»དབོན།
«སྣར་»མང་པོ། ●«གཡུང་»བདུངས། «པེ་»བདུབས། «སྣར་»བཏུབས། • «щ»«»
«ཞོལ་»ཚུན་ཆད། 《སྣར་»ཕན་ཆད། ●«སྣར་»-ཆེན་པོ།
༦༢༣ ●«སྣར་»-ཆེན་པོ། ●«སྣར་»ནས། ●«སྣར་»-ཀྱང་། ●«སྣར་»སོ། ●《སྣར་»-སྨྲས་པ།
●«གཡུང་>«པེ་»བགྱེས་ཞེས་ཞུས། 《སྣར་»དགྱེས་ཞེས་དྲིས། ●«གཡུང་»«པེ་»+པ། 《སྣར་»པའི་
སྡིག་པ། ●«སྣར་»པ་འདི། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»-སུས། 《སྣར་»«ཁུ་»ནས། སུས།
●《སྣར་》སྨྲས། ● «གཡུང་》ཏ་ལི་བུ་ཏྲ་སྟེ་དམར་བུ་ཅན་ཞེས་བྱ།
