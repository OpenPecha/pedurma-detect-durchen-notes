འཕགས་པ་ཡངས་པའི་གྲོང་ཁྱེར་དུ་འཇུག་པའི་མདོ་ཆེན་པོ།

བསྡུར་མཆན།
༢༣༥ ●མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད།། ●«སྣར་»བི་པུ་ལེ། «ཞོལ་»བི་
ཤཱ་ལཱི། · ●«གཡུང་»«པེ་»+ཡ་ན། 《ལི་»«ཅོ་»+ཡཱ་ན། ●«སྣར་»+ཏྲ་ན། ●«ཞོལ་»པའི།
●«ལི་»ཅིག ●«སྣར་»«ཞོལ་»བར། ●«ལི་»«པེ་»«ཅོ་»ཀྱིས། ●«སྣར་»ཏེ། ●«གཡུང་»
«ལི་»བྲི་ཛིར་ཡུལ། 《སྣར་»བྲི་ཛིར། «ཞོལ་»-ཡུལ། ●«ཁུ་»ཕྱིར། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»
མྲ་བསྲུང་། 《སྣར་》སྨྲ་སྲུང་།
༢༣༦ ●«ལི་»«ཅོ་»ཏེ། ●«གཡུང་»ཀྱི། ●«ལི་»ཀྱི། ●«སྣར་»«ཞོལ་»+ལྷ་མ་ཡིན་ཐམས་ཅད་ཀྱིས་
བསམས་པ། ●«པེ་»གཞག ●«སྣར་»«ཞོལ་»པ། ●«གཡུང་»མུན་ཙ་ཏ། མུན་ཙ་ཏ། 《«པེ་»མུན་
ཙ་ཏ། མུན་ཙ་ཏ། མུན་ཙ་ཏ། མུན་ཙ་ཏ། 《སྣར་»-མུཉྩ་ཏ། _ མུཉྩ་ཏ། • «»
《ལི་»«པེ་》《སྣར་》《ཅོ་»ཤིག ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»རྣམས། ●«གཡུང་»«པེ་»ནིར་་་་་
གད་ཚ། ( ཆོས་ཚན་འདིའི་ནང་དུ་ནིརྒཙྪ་ཏ་ཞེས་པ་རྣམས་«གཡུང་»«པེ་»ལ་ནིར་གད་ཚ་ཞེས་འཁོད་ཀྱང་
གཤམ་དུ་བསྡུར་མཆན་དུ་བཀོད་མེད།) ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»-ཡང་།
●«གཡུང་»«ལི་»
«པེ་»«ཅོ་»ཤིག ●«གཡུང་»-དང་། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»པའི་ལྷ།
●«གཡུང་»«ལི་》
«པེ་»«ཅོ་»-དབང་བདག་དང་བཅས་པ་རྣམས་དང་། ●«ལི་»གྱིས།
༢༣༧ ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»-དེ་དག ●«ཞོལ་»གི། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»པ་དང་།
●«གཡུང་»«ལི་»«པེ་»«ཅོ་»པ། ●«ཞོལ་»བསྲུང་། ●«སྣར་»-སུ་མུ། ●«གཡུང་»«པེ་»-སུ་མུ་
རུ༑ ●«གཡུང་»«པེ་》«སྣར་»«ཞོལ་»+མུ་རུ། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»-མི་རི། ●«སྣར་»
རི་མི། ( ཆོས་ཚན་འདིའི་ནང་དུ་མི་རི་ཞེས་པ་རྣམས་«སྣར་»ལ་རི་མི་ཞེས་འཁོད་ཀྱང་གཤམ་དུ་བསྡུར་མཆན་དུ་
བཀོད་མེད།)
●«གཡུང་»«པེ་»-མི་རི། «ཞོལ་»+མི་རི། ●«གཡུང་»«པེ་»རཱི། ●«གཡུང་»ཏ
●«ཞོལ་»མཱི། ●«སྣར་»+ཏི། ●«སྣར་»ཀཾ་ཀ་ར། «ཞོལ་»ཀ་ཀ་ར། ●«གཡུང་»«པེ་»རཱ།
●«གཡུང་»«པེ་»ཀ་རཱ། «སྣར་»ཀཾ་ར་ཙ། ●«སྣར་»«ཞོལ་»+ཀཾ་ཀ་རཱ། ཀཾ་ཀ་རཱ། ཀཾ་ཀ་རཱ།
●«གཡུང་》རཱོ་རཱི། 《པེ་»«སྣར་»རཱོ་ཏཱི། ●«གཡུང་»ཤོང་། ●«ཞོལ་»རཱ། ●«སྣར་»+ཀཾ་ཀ་རི་ཤི།
●«གཡུང་»«པེ་》《སྣར་»«ཞོལ་»ཕུ། ( ཆོས་ཚན་འདིའི་ནང་དུ་རི་པུ་ཞེས་པ་རྣམས་«གཡུང་»«པེ་»
●«སྣར་»«ཞོལ་»+རི་ཕུ།
«སྣར་»«ཞོལ་»ལ་རི་ཕུ་ཞེས་འཁོད་ཀྱང་གཤམ་དུ་བསྡུར་མཆན་དུ་བཀོད་མེད།)
●«གཡུང་»-ཐ། ●«གཡུང་»ཏི། 《སྣར་》ཏེ། ●«གཡུང་»«པེ་»ས་མཱ་ཡ་ཐ། «སྣར་»པཱ་ཡ་ཡ་ཏ།
●«གཡུང་»པཱུ་རི་པཱུ། ས་མཱ། «སྣར་»ཕུ་རི་ཕུ། པ་ལ། «ཞོལ་»པུ་རི་པུ། པ་ལཱ།
