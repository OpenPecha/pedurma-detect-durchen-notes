མ་མ། མནྡྲ་བ་དཱ། «ཞོལ་>སཏྭ་ནཱན། སིད་དྷྱན་ཏུ་མཱ་མ། མནྡྲ་བ་དཱ། ●«སྣར་»གའི།
●«སྣར་»སྲུང་། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»-སྔགས། ●«གཡུང་»«པེ་»ཀྱི་བཙེ་བ། 《སྣར་»
ཀྱིས་གཙེས་པ།
༥༦༧ ●«སྣར་»-ལ། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»-པ། ●«སྣར་»-ཛ། ●«སྣར་»-ཧ། ●«སྣར་»
«ཞོལ་»ལེ། ●«གཡུང་»«པེ་»མྲི་ཏ་ག་ནེ། པུ་རུ་ཤ་ནེ། བེ། «ལི་»མྲྀ་ཏ་ག་ཎེ། པ་རུ་ཥ་ཎི། བེ།
«ཅོ་»མྲི་ཏ་ག་ཎེ། པ་རུ་ཥ་ཎི། བེ། «སྣར་»མྲི་ཏ་གྷ་ནེ། པ་རུ་ཤ་ནེ། བཻ། «ཁུ་»མྲྀཏ་གྷ་ནེ། བ་རུ་
ཤ་ཎེ། བཻ། «ཞོལ་»མྲི་ཏ་གྷ་ནེ། པ་རུ་ཤ་ཎེ། བཻ། ●«གཡུང་»«པེ་»བྷད་ཏེ། བྷི་དྷི་ཧི། «ལི་»
«ཅོ་»བཏྟཱི། བི་དྷི་ཧི། «ཞོལ་»བཏྟེ། བི་དྷི་ཧེ། ●«གཡུང་»«པེ་»ཨམ་མྲི་ལུ། 《སྣར་»ཨ་མྲི་ལུ།
«ཅོ་»ཨ་མྲི་ལུ། ●«ཞོལ་»ཀཻ། ●«གཡུང་»«ལི་»«ཅོ་»–བྷེ་ཛུ་ཡུ། 《སྣར་»བཻ་ཛུ་ཡུ། ●«གཡུང་»
«ལི་»«ཅོ་»བེེ། «པེ་»བེ། ●«སྣར་»གའི། ●«སྣར་»སྲུང་། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»
-སྔགས༑ ●«ཅོ་»གདོན། ●«སྣར་»གཙེས་པ། ●«གཡུང་»«ལི་»《སྣར་»«ཅོ་»-ལ། ●«གཡུང་»
«པེ་»«སྣར་»ནི། ●«སྣར་»ཀཱ་རི་ཀཱ་རི། 《ཞོལ་»ཀ་རེ་ཀ་རེ། ●«གཡུང་»«པེ་»བྷན་དྷ། ●«སྣར་»
གའི༑ ●«ལི་»«ཅོ་»རྙེད། ●«ལི་»«ཅོ་»བསྲུངས། 《སྣར་»སྲུང་། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»
-སྔགས༑ ●«ཁུ་»-མི། ●«ཞོལ་»ལས་། ●«སྣར་»«ཞོལ་»པ་སྟེ། ●«སྣར་»+གྱི། ●«སྣར་»
གཙེས་པ།
༥༦༨ ●«གཡུང་»«ལི་»«སྣར་»«ཅོ་»-ལ། ●«སྣར་»«ཞོལ་»ཞེས། ●«གཡུང་»«པེ་»ཙན་དྷ།
●«གཡུང་»«པེ་»མ་བྷད་ཏྲེ། 《ལི་»«ཅོ་»མ་བྷ་དྲེ། 《སྣར་»མ་བྷ་ཏེ། ●«ལི་»«ཅོ་»རྀ་ཤི་རྀ།
●«གཡུང་»«ལི་»«པེ་》«ཅོ་»གཱ། ●«གཡུང་»«པེ་»སཏྭ་ནཱན། 《སྣར་»«ཞོལ་»སཏྭ་ནན།
●«སྣར་»གའི། ●«སྣར་》སྲུང་། ●«གཡུང་»«ལི་》«པེ་»«ཅོ་»-སྔགས། ●«ཞོལ་»ལས།
●«སྣར་»གཙེས་པ། ●«གཡུང་»«ལི་》《སྣར་》«ཅོ་»-ལ། ●«གཡུང་»«ལི་»ཏེ། «པེ་»འདི་སྐྱེ་ཏེ།
«ཅོ་»སྐྱེ་སྟེ། ●«སྣར་»«ཁུ་»«ཞོལ་»སིདྡྷི་སུ་སིདྡྷི། ●«གཡུང་»«པེ་》《སྣར་»«ཞོལ་»-མུཀྟེ།
●«གཡུང་»«པེ་»མུག་ཏེ། ●«གཡུང་»རན་ནྱ་གཏྗེ། «པེ་»རན་ནྱ་གཏྗེ། རཏྣ་གནྟེ། 《སྣར་»རེ་ཎྱ་
གཎྷེ༑ རཏྣ་གཎྷེ། ●«གཡུང་»ཏྲེ། «པེ་»ཏྲ། ●«གཡུང་»«པེ་»སརྦ་ཨནྟྲ་སཱ་དྷ་ཏྲེ། མན་སི། 《ལི་»
«ཅོ་»སརྠཱ་རྦ། སཱ་དྷ་ནི། མ་ན་སི། 《སྣར་»སརྦ་ཨརྠ་ས་དྷ་ནི། མ་ས་ནི། «ཞོལ་»སརྦ་ཨརྠཱ། ས་དྷ་ནི།
མ་ན་སི། ●«གཡུང་»«པེ་»ཨད་ཏྱད་བུ་ཏེ། ཨ་ར་ཛེ་ཨད་ཙུ། ཨམཱི། «ལི་»«ཅོ་»ཨནྡྷུ་ཏེ། ཨ་ཏྱནྡྷ།
«སྣར་»ཨཏ་བྷུ་ཏེ། ཨ་ཏྱ་བྷུ། ●«གཡུང་»«པེ་»པྲ་དྨ་པྲ་དྨ་ས། 《ལི་»«ཅོ་»ཨུཙྪུ་ཏེ། ཨ་མྲྀ་ཏེ། ཨ་ར་
