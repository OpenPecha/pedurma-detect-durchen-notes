བཀའ་འགྱུར། རྒྱུད་འབུམ། ༼༽ ༨༩
དཾ་དྷ་སི་དྷ་རི་དུ་ན། ཨིད་ཚ་ས་དེ་བོ་བར་ཤ་ཏི་བྷྱ་བྷོ་ཏ་ཡ་ཏི་གྲཱ་ཐ་ཏ། «ཞོལ་»-བིན་དྭ་ན། ●«གཡུང་»རཱ་
ཛེ། ས་མེ་དྷི་བྷར་ཏ་ཡན་ཏུམ། 《ལི་»«ཅོ་»རཱ་ཛ་ས་མཱ་དྷི་བནྡྷ་ཏ་ཡ་ན་ཏུཾ། 《པེ་»རཱ་ཛེ། ས་མཱ་དྷི་བྷར་དྷ་
ཏ་ཡ་ན་ཧཱུཾ། 《སྣར་»ར་ཛེ། ས་མ་དྷི་བྷ་ར་ཏ་ཡ་ཏུཾ། 《ཞོལ་»རཱ་ཛ། ས་མཱ་དྷི་བཌྷ་ཏ་ཡན་ཏུཾ། ●«སྣར་»
བ༑ ●«སྣར་»བྷོ། ●«གཡུང་»«པེ་》《སྣར་»«ཞོལ་»བྷེ། ●«གཡུང་»གི་གྲ་ཧ་དེ་བ་ཨཱ་ནན་ཏ་སྱ་གྲི་ཧ་
ནི་ཡང་། ཨ་ཀ་མ་ན། «པེ་»གི་གྲ་ཧ་དེ་བ་ཨཱ་ནན་ཏ་སྱ་གྲི་ཧ་ནི་ཡཱ། ཨ་ཀ་མ་ན། 《སྣར་»གི་གྲ་ཧ་དེ་བ་ཨཱ་
ནནྟ་སྱ་གྲེ་ཧ། ནི་ཡཾ་ཨ་ཀ་མ་ན། «ཁུ་»ཁི་གྲ་ཧཱ་དེ་བ། ཨཱ་ནནྡ་སྱ།...... ག་མ་ནཱ། «ཞོལ་»ཁི་གྲ་ཧ་དེ་བ།
ཨཱ་ནནྟ་སྱ། གྲ་ཧཱ་ཎི་ཡ། ཨ་ག་མ་ན། ●«གཡུང་»«སྣར་»ན་ཡ། ཨ་ག་མ་ན། ●«གཡུང་»«སྣར་»ན་ཡ། 《པེ་»ཎཱི་ཡ། «ཞོལ་»ཎི་ཡཱ།
●«གཡུང་》《སྣར་»-ཧ། ●«གཡུང་»བདུན། ●«གཡུང་»«པེ་»«སྣར་»«ཞོལ་»ལྕིས། ●«གཡུང་»
«པེ་»བྲིས་ཏེ། དུར་བའི་གཅལ། 《སྣར་»བྲིས་ཏ་ཀུ་ཤ་ཆལ་པར། «ཅོ་»བྲིས་ཏེ། དཱུར་བའི་གཅལ།
●«སྣར་»ནས། ●«སྣར་»«ཞོལ་»གའི། ●«སྣར་»«ཞོལ་»ལ་སྔགས་བཟླས་ཤིང་། ●«ལི་»«སྣར་»
«ཅོ་»བསྲེག ●«སྣར་»«ཞོལ་»པ་དང་། ●«སྣར་»«ཞོལ་»-འགྲུབ་པར། ●«སྣར་»ཏུ། ●«སྣར་»
«ཞོལ་»+ཆེན་པོ། ●«གཡུང་»«ལི་》《པེ་》《ཅོ་》སྟོ། ●《གཡུང་»«པེ་»ཆེན་པོ་དི། 《སྣར་»དྷི།
●«སྣར་»«ཞོལ་»-དགེ་སློང་། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»གཏན་ལ་ཕབ་པའོ།
《སྣར་»བཅོས་ཤིང་
བསྒྱུར་བའོ། ●«ལི་»«ཅོ་»+ཞུས་ཏེ་དག་གོ། །ཡང་ཞུས་ཤེ་སྐར་ཞུས་ཐང་རྒྱལ་གཉིས་ཀྱི་མ་དཔེ་གསུམ་ལ་
གཏུགས་ཏེ་ཞུས་ཆེན་བགྱིས་བདོག་ཅིང་དག་པར་བྱས་པས་སླན་ཆད་ཀྱང་འདི་ལྟར་བྲིས།།
