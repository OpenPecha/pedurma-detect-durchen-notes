ཀེ་བེེ་ལུ་ཀེ་ཨ་མྲྀ་ཏ་ག་ཎི་པུ་རུ་ཥ་ཎི་བེ་རུ་ཏེ་ཀེ་བ་རུ་བད་ཏ། བི་དྷེ་ཧེ་ཀེ་བིན་ད་པི་ལེ་ནི། ཨ་མྲི་ལུ་ཏེ་ཀི། བཻ་
ཛུ་ཡུ་བཻ་ཛུ་ཡུ། 《སྣར་»ཛ་ཧ་ཛ་ཧ་ཛ་རུ་ཛ་ཧ་རུ། བཻ་དུ་ཀེ་བེེ་དུ་ཀ ཨ་མྲྀ་ཏ། གྷ་ནེ། བུ་རུ་ཥ་ཎི། བཻ་
དུ་ཥ་ཎི། བཻ་དུ་ཏི་ཀེ་བ་རུ་བཏྟ། བི་དྷི་ཧི་ཀེ། བྷ་ནྡི་བ་རི་ནི། ཨ་མྲི་ཏི་ཀེ། བ་ཧུ་ཛུ་ཡུ། «ཞོལ་»ཛ་ཧ་
ཛ་ཧ་རུ། ཛ་ཧ་ཛ་ཧ་ཛ་ཧ་རུ། བཻ་ལུ་ཀེ་བེེ་ལུ་ཀེ། ཨ་མྲྀ་ཏ་ག་ཎིཾ། པུ་རུ་ཥ་ཎི། བཻ་རུ་ཏེ་ཀེ་བ་རུ་བཏྟ། བི་
དྷ་ཧེ་ཀེ། བིནྡ་བི་ལེ་ནི། ཨ་མྲྀ་ལུ་ཏེ་ཀི། བཻ་ཛུ་ཡུ་བཻ་ཛུ་ཡུ། ●«སྣར་»གའི། ●«གཡུང་»«པེ་»ཀྱིས།
●«གཡུང་》《ལི་》《པེ་»«ཅོ་»-རྣམས། ●《སྣར་》སྲུང་། ●《སྣར་»«ཞོལ་»+པ་ཡིན། ●«སྣར་»
«ཁུ་»«ཞོལ་»+ཞིག་གིས། ●«སྣར་»«ཞོལ་»+ཁ་ཏོན་དུ། ●«སྣར་»ཡོངས་སུ་འཛིན་པར་བྱེད་ན།
«ཞོལ་»ཡོངས་སུ་འཛིན་པར་བྱེད་ན་འཇིགས་ཤིང་སྐྲག་པ་རྣམས་ལས་ཐར་བ་དང་། ●«ཞོལ་»+མི་དང་།
●«སྣར་»པས་ཕྱིར་རྒོལ་བ་དང་། གདོན་བགེགས། ●«སྣར་»གྱིས་འཚོ་བའི་འཇིགས་པ་ལས་ཐར་བར་
འགྱུར་ཞིང་། ●«སྣར་»«ཞོལ་»ནས། ●«སྣར་»«ཞོལ་»ཅིང་རྗེས་སུ་དྲན་པར་འགྱུར་རོ། ●«སྣར་»
པོས༑ «ཞོལ་»+རྣམས་ཀྱིས། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»ལ་གཟུངས། «སྣར་»«ཞོལ་»-གཟུངས།
●«སྣར་»«ཞོལ་»གཟུངས་གང་ཡིན་པ་དེ།
༡༩༧ ●«སྣར་»«ཞོལ་»གཏེར་ཞེས་བྱའོ། ●«སྣར་»«ཞོལ་»-གཟུངས་ལ། ●«གཡུང་»«པེ་»ནི། བི་ཏེ་བི།
●«སྣར་»+ཧེ་རུ། ●«གཡུང་»«པེ་»ར་ཙུ་བ་ཏི་མི། «ལི་»ར་ཙུ་རུ་བ《ནྡྷ་ནི། «ཞོལ་»རུ་ཅུ་རུ། བ《ནྡྷ་
ནི༑ ●«སྣར་»གའི། ●«སྣར་»«ཞོལ་»+རྣམས། ●«སྣར་»གཟུང། ●«སྣར་»«ཞོལ་»+པ་ཡིན།
●«གཡུང་»«ལི་»«པེ་»«ཅོ་»-ཞིག་གིས། ●«སྣར་»«ཞོལ་»+ཁ་ཏོན་ཏུ། ●«ཞོལ་»+ཡོངས་སུ།
●«སྣར་»«ཞོལ་»+པར་བྱེད། ●«སྣར་»-འཇིགས་ཤིང་སྐྲག་པ་རྣམས་ལས་ཐར་བ་དང་། ●«ཞོལ་»
དྭགས༑ ●«སྣར་»པས་ཕྱིར་རྒོལ་བ་དང་། གདོན་བགེགས། ●«སྣར་»གྱིས་འཚེ་བའི་འཇིགས་པ་རྣམས་
ལས་ཐར་བར་འགྱུར་ཞིང་། «ཞོལ་»གྱི་རྣམ་པ་དང་། ●«སྣར་»«ཞོལ་»ནས། ●«ལི་»རྗེད་པར་དྲན་ནོ།
«སྣར་»«ཞོལ་»བརྗེད་ཅིང་རྗེས་སུ་དྲན་པར་འགྱུར་རོ། «ཁུ་»བརྗེད་པར་བྲན་ནོ། ●«སྣར་»པོས།
«ཞོལ་»+རྣམས་ཀྱིས། ●«གཡུང་»«ལི་»«པེ་»«ཅོ་»ལ་གཟུངས། 《སྣར་»«ཞོལ་»-གཟུངས།
●«སྣར་»«ཞོལ་»གཟུངས་གང་ཡིན་པ་དེ་ནི། ཚད། ●«སྣར་»«ཞོལ་»བྱའོ། ●«སྣར་»«ཞོལ་»
-གཟུངས་ལ། ●«གཡུང་»«པེ་»རེ་ཙཎ་ཌ་ལི། ཀེ་རུ་ཀུ་རམ་བྷ་རད་ཏེ་ཏོ་ར་སེ་པད་ཏ། པད་ཏ་བ་སེ།
རི༑ 《སྣར་»རི་ཙཎྜཱ་ལི་ཀེ། ཀུ་ལ་མཱ་བྷ་ཏ། ཏོ་ར་སི། བ་ཏ་བ་ཏ་སི། 《ཞོལ་»རི་ཙཎྜཱ་ལི་ཀེ། ཀུ་ལ་མ་
བྷ་དྲི། ཏོ་ར་སི་པ་ཏ་པ་ཏ་སི། ●«གཡུང་»ཀ་ཤི་ཀ་བི་ཤི་རི། སྭ་སིད་དྷི། སརྦ་ས་ཏྭ། «པེ་»ཀ་ཤི་ཀ་བི་
ཤི་རི། སྭ་སིད་དྷི། སརྦ་སཏྭ་ནན། 《སྣར་»ཀ་ཤི་རི། ཀ་བི་ཤི་རི། སིདྡྷི། སརྦ་ས་ཏྭ་ནཱན། «ཞོལ་»
ག་ཤི་རི་གཱ་བཾ་ཤི་རི་སྭསྟི། སརྦ་སཏྭ་ནཱན། ●«སྣར་»གའི། ●«སྣར་»རྣམས་སྲུང་། 《ཞོལ་»རྣམས་
བསྲུང་། ●«སྣར་»«ཞོལ་»+པ་ཡིན། ●«གཡུང་»«པེ་»གཟུང་སྔགས། «ལི་»«ཅོ་»–ཞིག་གིས།
《སྣར་》-སྔགས། ●«སྣར་»«ཞོལ་»+ཁ་ཏོན་དུ ●«སྣར་»«ཞོལ་»ཡོངས་སུ་འཛིན་པར་བྱེད།
●«སྣར་»-གཇིགས་ཤིང་སྐྲག་པ་རྣམས་ལས་ཐར་བ་དང་། ●«ཞོལ་»དྭགས། ●«སྣར་»པའི་ཕྱིར་རྒོལ་བ་
